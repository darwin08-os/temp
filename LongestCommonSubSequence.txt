#include<stdio.h>
#include<string.h>

void main(){

    int i,c[100][100],j,l1,l2;
    char x[100],y[100],sol[100];

    printf("\nEnter sequence 1 : ");
    scanf("%s",&x);
    l1 = strlen(x);

    //printf("%d",l1);

    printf("\nEnter sequence 2 : ");
    scanf("%s",&y);
    l2 = strlen(y);

    for(i=0;i<=l1;i++){
        for(j=0;j<=l2;j++){
            if(i==0 || j==0)
            {
                c[i][j] = 0;
            }
            else if(x[i-1] == y[j-1]){
                c[i][j] = 1+c[i-1][j-1];
            }
            else{
                if(c[i-1][j] > c[i][j-1]){
                    c[i][j] = c[i-1][j];
                }
                else{
                    c[i][j] = c[i][j-1];
                }
            }
        }
    }

    for(i=0;i<l1;i++){
        for(j=0;j<l2;j++){
            printf("%4d",c[i][j]);
        }
        printf("\n");
    }


    //backtracking
    i = l1;
    j = l2;
    int index = c[l1][l2];
    sol[index] = '\0';

    while(i>0 && j>0){
        if(x[i-1] == y[j-1]){
            sol[index-1] = x[i-1];
            i--;
            j--;
            index--;
        }
        else if(c[i-1][j] > c[i][j-1]){
            i--;
        }
        else{
            j--;
        }

    }

    printf("LCS : %s",sol);

}
